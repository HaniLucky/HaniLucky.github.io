        # 查询表结构

	table_name 表名,
	column_name 列名,
	data_type 数据类型,
	CHARACTER_MAXIMUM_LENGTH 长度,
	IS_NULLABLE 是否为NULL,
	column_comment 备注
FROM
	information_schema.COLUMNS 
WHERE
	table_schema = 'cs' 
	AND table_name = 'user'
	ORDER BY ORDINAL_POSITION;
