                # 查询指定表的结构

                select col.column_name,
	   case when col.data_type = 'integer' then 'INT4' 
	   when col.data_type = 'smallint' then 'INT2' 
	   when col.data_type = 'bigint' then 'INT8'
	   when col.data_type = 'character varying' then 
	   	'VARCHAR(' || col.character_maximum_length || ')'
	   when col.data_type = 'timestamp without time zone' then 'TIMESTAMP'
	   when col.data_type = 'numeric' then
	   	'NUMERIC(' ||col.numeric_precision || ',' || col.numeric_scale|| ')'
	   when col.data_type = 'numeric' then
	   	'DECIMAL(' ||col.numeric_precision || ',' || col.numeric_scale|| ')'
	   when col.data_type = 'double precision' then 'DOUBLE'
	   else col.data_type 
	   end as data_type,
		 '',
      case when col.is_nullable = 'YES' then '是' else '否' end as is_nullable,
-- 	   col.character_maximum_length,
-- 	   col.numeric_precision || ',' || col.numeric_scale,
       col.column_default,
       des.description
  from information_schema.columns col
  left join pg_description des
    on col.table_name ::regclass = des.objoid
   and col.ordinal_position = des.objsubid
 where table_schema = 'cs'
   and table_name = 'rp_zk_reconciliation_report'
 order by ordinal_position;

                # 创建序列

                -- 创建一个名为 bg_interface_invoke_log_id_seq 的序列 开始值为 82
CREATE SEQUENCE bg_interface_invoke_log_id_seq
  START WITH 82
  INCREMENT BY 1
  NO MINVALUE
  NO MAXVALUE
  CACHE 1;

                -- 将 bg_interface_invoke_log_id_seq 序列绑定到 bg_interface_invoke_log 表的 id 列上
