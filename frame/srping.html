# 直接在类的成员方法上直接添加事务是否生效

生效,事务直接回滚

@Service("userServiceImpl2")
public class UserServiceImpl2 {
    private final Logger logger = LoggerFactory.getLogger(UserServiceImpl2.class);
    @Autowired
    UserDao userDao;
    @Transactional(rollbackFor = RuntimeException.class)
    public void t1() {
        User user = new User();
        user.setName("HLJ");
        userDao.insert(user);
        Objects.requireNonNull(null);
    }
}
